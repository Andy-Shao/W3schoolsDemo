The typeof operator

var x = 5;
var y = "John";
typeof x // Returns number
typeof y // Returns string
typeof 3.14 // Returns number
typeof false // Returns boolean
typeof [1,2,3,4] // Returns object
typeof {name:'john', age:34} // Returns object
